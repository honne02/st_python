    exchange = True
    k = 0
    step = 0
    for i in range(0, dim - 1):

        exchange = False
 
        for j in range(len(arr) - 1, step, -1):
            if arr[j] < arr[j - 1]:
                exchange = True
                arr[j-1], arr[j] = arr[j], arr[j-1]
                k = j
        step = k
